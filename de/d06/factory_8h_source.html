174template< class Cre,
175 class I = const char *,
176 class Com = less< I > >
177class Factory {
178public:
179 typedef Cre Creator;
180 typedef I Index;
181 typedef Com Compare;
182private:
183 typedef std::map< Index, std::stack< Creator >, Compare > Map;
184private:
185 Map M_creators;
186
187public:
191 Factory()
192 { }
193
197 ~Factory()
198 { }
199
205 void reg( Creator c,
206 const Index & idx )
207 {
208 M_creators[ idx ].push( c );
209 }
210
215 void dereg( const Index & idx )
216 {
217 //std::cerr << "dereg [" << idx << "]" << std::endl;
218 typename Map::iterator i = M_creators.find( idx );
219 if ( i != M_creators.end() )
220 {
221 if ( ! i->second.empty() )
222 {
223 i->second.pop();
224 }
225
226 if ( i->second.empty() )
227 {
228 M_creators.erase( i );
229 }
230 }
231 }
232
239 bool getCreator( Creator & c,
240 const Index & idx ) const
241 {
242 typename Map::const_iterator i = M_creators.find( idx );
243 if ( i != M_creators.end()
244 && ! i->second.empty() )
245 {
246 c = i->second.top();
247 return true;
248 }
249 return false;
250 }
251
256 std::list< Index > list() const
257 {
258 std::list< Index > rval;
259 for ( typename Map::const_iterator i = M_creators.begin();
260 i != M_creators.end();
261 ++i )
262 {
263 rval.push_back( i->first );
264 }
265 return rval;
266 }
267
273 std::ostream & printList( std::ostream & o = std::cout ) const
274 {
275 for ( typename Map::const_iterator i = M_creators.begin();
276 i != M_creators.end();
277 ++i )

280 << "(" << i->second.size() << ")"
281 << std::endl;
282 }

284 }

300 size_t size( const Index & idx ) const
301 {
302 for ( typename Map::const_iterator i = M_creators.begin();
303 i != M_creators.end();
304 ++i )
305 {
306 if ( ! Compare()( idx, i->first )
307 && ! Compare()( i->first, idx ) )
308 {
309 return i->second.size();
310 }
311 }
312 return 0;
313 }
