229class BallControllableDistancePlayerEvaluator
230 : public PlayerEvaluator {
231
232private:
233 SideID M_our_side;
234
235public:
236 BallControllableDistancePlayerEvaluator( const WorldModel & wm )
237 : M_our_side( wm.ourSide() )
238 { }
239
245 double operator()( const AbstractPlayerObject & p ) const
246 {
247 const double penalty_area_abs_x
248 = ServerParam::i().pitchHalfLength() - ServerParam::i().penaltyAreaLength();
249
250 if ( p.goalie()
251 && p.pos().absY() < ServerParam::i().penaltyAreaHalfWidth()
252 && ( ( p.side() == M_our_side && p.pos().x <= (- penalty_area_abs_x) )
253 || ( p.side() == NEUTRAL && p.pos().absX() <= penalty_area_abs_x )
254 || ( p.pos().x >= (+ penalty_area_abs_x) ) ) )
255 {
256 return ServerParam::i().catchableArea();
257 }
258
259 return ( p.playerTypePtr()
260 ? p.playerTypePtr()->kickableArea()
261 : ServerParam::i().defaultKickableArea() );
262 }
263};
