104 const AngleDeg & normalize()
105 {
106 if ( M_degree < -360.0 || 360.0 < M_degree )
107 {
108 M_degree = std::fmod( M_degree, 360.0 );
109 }
110
111 if ( M_degree < -180.0 )
112 {
113 M_degree += 360.0;
114 }
115
116 if ( M_degree > 180.0 )

118 M_degree -= 360.0;
119 }
120
121 return *this;
122 }
