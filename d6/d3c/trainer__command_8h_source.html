1021/*
1022
1023----------
1024Offline Coach Initialization Command
1025See rcssserver/src/netif.C : void Coach::parse_init
1026
1027"(init (version VERSION))"
1028if VERSION is illegal
1029-> "(error illegal_command_form)"
1030else
1031-> "(init ok)"
1032
1033receive parameter info
1034receive changed players info
1035----------
1036
1037----------
1038
1039----------
1040Offline Coach Command List
1041See rcssserver/src/netif.C : void Coach::parse_command(const char *command)l
1042
1043
1044if use irregal character for command string
1045-> "(error illegal_command_form)"
1046else if irregal command
1047-> "(error unknown_command)"
1048
1049----------
1050"(start)" : start game (change playmode before_kickoff to kick_off_?)
1051-> return "(ok start)"
1052----------
1053"(change_mode NEW_PLAYMODE)" : change playmode
1054if NEW_PLAYMODE string is NULL
1055-> "(error illegal_command_form)"
1056else NEW_PLAYMODE is not valid
1057-> "(error illegal_mode)"
1058else
1059-> "(ok change_mode)"
1060----------
1061"(move OBJ_NAME X Y ANG VX VY)" : move pos, set ang, vel
1062"(move OBJ_NAME X Y ANG)" : move pos, set ang
1063"(move OBJ_NAME X Y)" : move pos
1064if (X,Y,ANG,VX,VY is not a number)
1065-> "(error illegal_command_form)"
1066else if (arg != 2 || arg != 3 || arg != 5)
1067-> "(error illegal_command_form)"
1068else
1069-> "(ok move)"
1070----------
1071"(look)" : get all movable objects' positional information
1072-> "(ok look TIME GOAL_INFO BALL_INFO PLAYER_INFO ...)"
1073----------
1074"(team_names)" : get team name
1075-> "(ok team_names (team l TEAM_NAME) (team r TEAM_NAME))"
1076----------
1077"(recover)" : recover all player's stamina, effort and recover
1078-> "(ok recover)"
1079----------
1080"(check_ball)" : check ball positional state
1081-> (ok check_ball BALL_POS_INFO)
1082BALL_POS_INFO :- "in_field" | "goal_l" | "goal_r" | "out_of_field"
1083----------
1084"(say MESSAGE)"
1085if MESSAGE is NULL string
1086-> "(error illegal_command_form)"
1087else
1088-> "(ok say)"
1089MESSAGE is [-0-9a-zA-Z ()./+*?<>_]
1090----------
1091"(ear ONOFF)" : turn on/off to get audio info
1092if ONOFF is NULL
1093-> "(error illegal_command_form)"
1094else if ONOFF_MODE == "on"
1095-> "(ok ear on)"
1096else if ONOFF_MODE == "off"
1097-> "(ok ear off)"
1098else
1099-> "(error illegal_command_form)"
1100----------
1101"(eye ONOFF)" : turn on/off to get visual info
1102if ONOFF is NULL
1103-> "(error illegal_command_form)"
1104else if ONOFF_MODE == "on"
1105-> "(ok eye on)"
1106else if ONOFF_MODE == "off"
1107-> "(ok eye off)"
1108else
1109-> "(error illegal_command_form)"
1110----------
1111"(change_player_type TEAM_NAME UNUM PLAYER_TYPE_ID)" : change hetero player type
1112if TEAM_NAME is not match
1113-> "(warning no_team_found)"
1114else if PLAYER_TYPE_ID is out of range (ID<0 or PlayerTypes<ID)
1115-> "(error out_of_range_player_type)"
1116else if UNUM is not match
1117-> "(warning no_such_player)"
1118else
1119-> "(ok change_player_type TEAM_NAME UNUM PLAYER_TYPE_ID)"
1120----------
1121"(done)" : thinking end nortification for sync mode
1122----------
1123"(compression LEVEL)" : set zlib compression level
1124if LEVEL is NULL string
1125-> "(error illegal_command_form)"
1126else if not HAVE_LIBZ
1127-> "(warning compression_unsupported)"
1128else if (LEVEL < 0 || 9 < LEVEL)
1129-> "(error illegal_command_form)"
1130else
1131-> "(ok compression LEVEL)"
1132----------
