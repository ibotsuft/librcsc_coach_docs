893 const PlayerObject * getFirstPlayer( const PlayerObject::Cont & players,
894 const int count_thr,
895 const bool with_goalie ) const
896 {
897 for ( PlayerObject::Cont::const_reference p : players )
898 {
899 if ( ! with_goalie
900 && p->goalie() )
901 {
902 continue;
903 }
904
905 if ( ! p->isGhost()
906 && p->posCount() <= count_thr )
907 {
908 return p;
909 }
910 }
911 return nullptr;
912 }

1463 template < typename REGION >
1464 bool existPlayerIn( const REGION & region,
1465 const PlayerObject::Cont & players,
1466 const int count_thr,
1467 const bool with_goalie ) const
1468 {
1469 for ( PlayerObject::Cont::const_reference p : players )
1470 {
1471 if ( p->posCount() > count_thr
1472 || p->isGhost() )
1473 {
1474 continue;
1475 }
1476 if ( p->goalie() && ! with_goalie )
1477 {
1478 continue;
1479 }
1480 if ( region.contains( p->pos() ) )
1481 {
1482 return true;
1483 }
1484 }
1485 return false;
1486 }
